# 双向链表
```
//节点类
class Node{
    constructor(data){
        //data为当前节点所存储的数据
        this.data = data;
        //next指向下一个节点
        this.next = null;
        //prev指向前一个节点
        this.prev = null;
    }
}

//双向链表类
class DoublyLinkedList{
    constructor(){
        //双向链表开头
        this.head = null;
        //双向链表结尾
        this.tail = null;
    }
}
```
